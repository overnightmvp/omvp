import { motion } from 'framer-motion'

interface AntiCardProps {
  id: string
  label: string
  selected: boolean
  onClick: (id: string) => void
}

export function AntiCard({ id, label, selected, onClick }: AntiCardProps) {
  return (
    <motion.button
      onClick={() => onClick(id)}
      className={`w-full p-4 rounded-lg border-2 text-left transition-smooth ${
        selected
          ? 'bg-accent2/10 border-accent2'
          : 'bg-card border-border hover:border-border-md'
      }`}
      whileHover={{ scale: 1.01 }}
      whileTap={{ scale: 0.99 }}
    >
      <div className="flex items-center">
        <span className="mr-2">{selected ? '✗' : '○'}</span>
        <div className="font-medium">{label}</div>
      </div>
    </motion.button>
  )
}

### Task 5: Create Quiz Save API Route

**File:** `src/app/api/quiz/save/route.ts`

Save quiz responses to Supabase after signup:

```typescript
import { createClient } from '@/lib/supabase/server'
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'
import { calculateAuthorityScore } from '@/lib/quiz/authority-algorithm'

export async function POST(request: NextRequest) {
  const supabase = createClient()

  // Verify user is authenticated
  const {
    data: { user },
  } = await supabase.auth.getClaims()

  if (!user) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  try {
    const body = await request.json()

    // Calculate authority score
    const authorityScore = calculateAuthorityScore(body)

    // Save quiz response to database
    const { data, error } = await supabase
      .from('quiz_responses')
      .upsert({
        user_id: user.id,
        name: body.name,
        handle: body.handle,
        primary_platform: body.primaryPlatform,
        secondary_channels: body.secondaryChannels || [],
        niche: body.niche,
        niche_category: body.nicheCategory,
        offers: body.offers || [],
        google_presence: body.googlePresence,
        ai_presence: body.aiPresence,
        website_status: body.websiteStatus,
        brand_tones: body.brandTones || [],
        anti_vision: body.antiVision || [],
        anti_custom: body.antiCustom,
        blocker: body.blocker,
        timeline: body.timeline,
        context: body.context,
        authority_score: authorityScore,
      })
      .select()
      .single()

    if (error) throw error

    // Also update profile with name and handle
    await supabase.from('profiles').upsert({
      id: user.id,
      name: body.name,
      handle: body.handle,
    })

    return NextResponse.json({
      success: true,
      data,
      authorityScore,
    })
  } catch (error: any) {
    console.error('Error saving quiz response:', error)
    return NextResponse.json(
      { error: error.message || 'Failed to save quiz response' },
      { status: 500 }
    )
  }
}
```

import { AuthForm } from '@/components/auth/AuthForm'
import { useSearchParams } from 'next/navigation'
import { useEffect } from 'react'
import type { Metadata } from 'next'

export default function SignupPage() {
  const searchParams = useSearchParams()
  const quizCompleted = searchParams.get('quiz_completed')

  useEffect(() => {
    // Check if quiz was completed
    if (quizCompleted && typeof window !== 'undefined') {
      const quizState = localStorage.getItem('quiz_state_v1')
      if (quizState) {
        // Store quiz state in sessionStorage for post-signup save
        sessionStorage.setItem('pending_quiz_save', quizState)
      }
    }

    <div>
      {quizCompleted && (

// In handleSubmit, after successful signup, add:

if (mode === 'signup') {
  const { data, error } = await supabase.auth.signUp({
    email,
    password,
    options: {
      emailRedirectTo: `${window.location.origin}/auth/confirm`,
    },
  })

  if (error) throw error

  // CRITICAL: Save quiz data if available
  const pendingQuizData = sessionStorage.getItem('pending_quiz_save')
  if (pendingQuizData) {
    try {
      const quizState = JSON.parse(pendingQuizData)

      // Save quiz data to database
      const saveResponse = await fetch('/api/quiz/save', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(quizState),
      })

      if (!saveResponse.ok) {
        throw new Error('Failed to save quiz data to database')
      }

      const savedData = await saveResponse.json()

      // VERIFICATION: Confirm data was saved correctly
      console.log('[Quiz Transfer] Data saved successfully:', {
        userId: data.user?.id,
        authorityScore: savedData.authorityScore,
        name: quizState.name,
        niche: quizState.niche,
      })

      // Clean up temporary storage
      sessionStorage.removeItem('pending_quiz_save')
      localStorage.removeItem('quiz_state_v1')

      console.log('[Quiz Transfer] localStorage and sessionStorage cleared')
    } catch (err) {
      console.error('[Quiz Transfer] CRITICAL: Failed to save quiz data:', err)
      // Keep data in sessionStorage for recovery
      setError('Quiz data may not have been saved. Please contact support.')
      return
    }
  } else {
    console.warn('[Quiz Transfer] No quiz data found in sessionStorage')
  }

  setSuccess(
    'Account created! Please check your email for verification link.'
  )

  // Redirect to YouTube connect after 2 seconds
  setTimeout(() => {
    router.push('/youtube-connect')
    router.refresh()
  }, 2000)
}
